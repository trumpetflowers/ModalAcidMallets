function initAudio() { if (ctx) return; ctx = new (window.AudioContext || window.webkitAudioContext)(); master = ctx.createGain(); master.gain.value = 0.5; delay = ctx.createDelay(); delay.delayTime.value = 0.375; const feed = ctx.createGain(); feed.gain.value = 0.3; delay.connect(feed); feed.connect(delay); master.connect(ctx.destination); delay.connect(master); } function playKick(t) { if(!S.pwr.drums) return; const o = ctx.createOscillator(); const g = ctx.createGain(); o.frequency.setValueAtTime(150, t); o.frequency.exponentialRampToValueAtTime(0.01, t+0.5); g.gain.setValueAtTime(S.params['drm-kick']/100, t); g.gain.exponentialRampToValueAtTime(0.001, t+0.5); o.connect(g); g.connect(master); o.start(t); o.stop(t+0.5); } function playAcid(t, f) { if(!S.pwr.acid) return; const o = ctx.createOscillator(); const flt = ctx.createBiquadFilter(); const g = ctx.createGain(); o.type='sawtooth'; o.frequency.setValueAtTime(f, t); flt.type='lowpass'; flt.frequency.setValueAtTime(S.params['acid-cut']*20, t); flt.Q.value = S.params['acid-res']/5; g.gain.setValueAtTime(0.2, t); g.gain.exponentialRampToValueAtTime(0.001, t+0.2); o.connect(flt); flt.connect(g); g.connect(master); if(S.params['fx-echo'] > 20) g.connect(delay); o.start(t); o.stop(t+0.2); } let nextT = 0; function scheduler() { while(nextT < ctx.currentTime + 0.1) { if(S.seq.drums[S.step]) playKick(nextT); if(S.seq.acid[S.step]) playAcid(nextT, 55 + (S.step%3)*12); updateUI(S.step); nextT += 60 / S.bpm / 4; S.step = (S.step + 1) % 16; } } function updateUI(s) { document.querySelectorAll('.step-btn').forEach(b => b.classList.remove('active')); document.querySelectorAll(`[data-idx="${s}"]`).forEach(b => b.classList.add('active')); } /** ── UI LOGIC (RETAINING YOUR DESIGN) ── **/ document.addEventListener('DOMContentLoaded', () => { // Build Grid const grid = document.getElementById('seq-grid'); ['mallet','acid','drums'].forEach(layer => { const row = document.createElement('div'); row.className = 'seq-row'; for(let i=0; i<16; i++) { const b = document.createElement('div'); b.className = 'step-btn'; if(S.seq[layer][i]) b.classList.add('on'); b.dataset.idx = i; b.onclick = () => { S.seq[layer][i] = !S.seq[layer][i]; b.classList.toggle('on'); }; row.appendChild(b); } grid.appendChild(row); }); // Knobs (Direct mapping to your DIV-based Knobs) document.querySelectorAll('.knob').forEach(k => { S.params[k.dataset.id] = k.dataset.val; k.style.transform = `rotate(${(k.dataset.val * 2.4) - 120}deg)`; k.onmousedown = (e) => { let startY = e.pageY; const move = (ev) => { let diff = startY - ev.pageY; let val = Math.min(100, Math.max(0, parseInt(S.params[k.dataset.id]) + diff)); S.params[k.dataset.id] = val; k.style.transform = `rotate(${(val * 2.4) - 120}deg)`; startY = ev.pageY; }; window.onmousemove = move; window.onmouseup = () => window.onmousemove = null; }; }); // Power Buttons document.querySelectorAll('.pwr-btn').forEach(b => { b.onclick = () => { S.pwr[b.dataset.t] = !S.pwr[b.dataset.t]; b.classList.toggle('active'); }; }); // Play/Stop document.getElementById('playBtn').onclick = (e) => { initAudio(); if(ctx.state === 'suspended') ctx.resume(); S.playing = !S.playing; if(S.playing) { nextT = ctx.currentTime; S.timer = setInterval(scheduler, 25); e.target.innerText = "STOP"; } else { clearInterval(S.timer); e.target.innerText = "START"; } }; });
